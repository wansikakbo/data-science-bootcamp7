ID =REGEXEXTRACT(A4:A8, "\d-\d{4}-\d{5}-\d{2}-\d")
Gender =IF(REGEXMATCH(A4:A8,  "M??s\."),"Female","Male")
Firstname =REGEXEXTRACT(A4:A8, "M.+\. ([A-Z]\w+) [A-Z]\w+")
Lastname =REGEXEXTRACT(A4:A8, "M.+\. [A-Z]\w+ ([A-Z]\w+)")
DOB =REGEXEXTRACT(A4:A8, "Date of Birth (.*)\n")
Age =YEAR(NOW())-RIGHT(F4:F8,5)
Address =REGEXEXTRACT(A4:A8, "Address (.*)\n")
Zipcode =RIGHT(H4:H8,5)
Expired Date =REGEXEXTRACT(A4:A8, "Expired Date (.*)")
Expired This Year =RIGHT(J4:J8,4)=YEAR(NOW())
